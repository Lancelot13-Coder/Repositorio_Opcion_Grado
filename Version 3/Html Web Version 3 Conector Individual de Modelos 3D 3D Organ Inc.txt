<script>
(async function(){
  
  const srcUrl = "https://3d-organ-inc-web.odoo.com/web/content/506?unique=1aae4ce1bd428c1987395b3cb6a6fc7c89a96f34&download=true";

  const statusEl = document.getElementById('viewer-status');
  const cont = document.getElementById('viewer-container');
  const mv = document.getElementById('mv');

  try {
    statusEl.textContent = 'Comprobando URL: ' + srcUrl;
    const resp = await fetch(srcUrl, { method: 'HEAD' }); 
    if (!resp.ok) {
      statusEl.innerHTML = '<strong>Error:</strong> La URL respondió con código ' + resp.status + '.<br>Revisa que la ruta sea accesible públicamente desde el navegador.';
      return;
    }

    
    const ac = resp.headers.get('access-control-allow-origin');
    if (ac && ac !== '*' && !location.origin.includes(new URL(srcUrl, location.href).origin)) {
      statusEl.innerHTML = '<strong>Advertencia CORS:</strong> el servidor responde con Access-Control-Allow-Origin = ' + ac + '<br>Esto puede bloquear la carga en model-viewer.';
    }

    mv.setAttribute('src', srcUrl);
    cont.style.display = 'block';
    statusEl.style.display = 'none';

  } catch (err) {
    statusEl.innerHTML = '<strong>Excepción:</strong> no se pudo obtener el archivo.<br>Revisa la consola para detalles.<br><pre style="white-space:pre-wrap;">' + err + '</pre>';
    console.error(err);
  }
})();
</script>
